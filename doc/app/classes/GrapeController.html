def edit 

  @grape = Grape.find(params[:id])
  @title = "Edit Grape: #{@grape.name}"
  if param_posted?(:grape)
    if @grape.update_attributes(params[:grape])
      flash[:notice] = "#{@grape.name} (#{@grape.id}) updated"
      redirect_to :action => "index"
    end
  end

end
